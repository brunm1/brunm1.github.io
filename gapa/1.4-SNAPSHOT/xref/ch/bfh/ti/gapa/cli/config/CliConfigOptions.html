3   import ch.bfh.ti.gapa.cli.config.model.ConfigField;
4   import org.apache.commons.cli.Option;
5   import org.apache.commons.cli.OptionGroup;
6   import org.apache.commons.cli.Options;
7   
8   /**
9    * Defines command line arguments for configuration.
10   */
11  public class CliConfigOptions extends Options {
12      public CliConfigOptions() {
13          Option websocketUri = Option.builder("w")
14                  .desc("Connect over websocket to this URI. Default: " + ConfigField.websocketUri.getDefaultValue())
15                  .hasArg()
16                  .argName("uri")
17                  .longOpt("websocket")
18                  .build();
19  
20          Option serverName = Option.builder("n")
21                  .desc("Name of the central server that receives and sends requests. Default: " + ConfigField.serverName.getDefaultValue())
22                  .hasArg()
23                  .argName("name")
24                  .longOpt("server-name")
25                  .build();
26  
27          Option config = Option.builder("c")
28                  .desc("Set the path to config file. Default: path to config.json in the same directory as executed jar")
29                  .hasArg()
30                  .argName("path")
31                  .longOpt("config")
32                  .build();
33  
34          this.addOption(websocketUri);
35          this.addOption(config);
36          this.addOption(serverName);
37      }
38  }
