1   package ch.bfh.ti.gapa.cli.config.reading.file.json;
2   
3   import ch.bfh.ti.gapa.cli.config.model.ConfigField;
4   import ch.bfh.ti.gapa.cli.config.model.CliInput;
5   import org.json.JSONObject;
6   
7   /**
8    * Reads the data from a json object into
9    * an {@link CliInput} instance.
10   */
11  public class JsonReaderImpl implements JsonReader {
12      /**
13       * Creates an instance of {@link CliInput} and copies data from a {@link JSONObject} to this instance.
14       * The websocketUri and serverName are read as string.
15       * The filters are read as {@link org.json.JSONArray}.
16       * All fields are optional.
17       */
18      public void read(CliInput cliInput, JSONObject jsonObject) {
19          cliInput.setWebsocketUri(jsonObject.optString(ConfigField.websocketUri.name(), null));
20          cliInput.setFilters(jsonObject.optJSONArray(ConfigField.filters.name()));
21          cliInput.setServerName(jsonObject.optString(ConfigField.serverName.name(), null));
22      }
23  }
