3   /**
4    * Defines exit codes with descriptions.
5    * If an exception happens in the cli application, the
6    * exception is wrapped and one of the following {@link GapaCommandLineExceptionType} is added.
7    */
8   public enum GapaCommandLineExceptionType implements CommandLineExceptionType {
9       INVALID_COMMAND_USAGE(5, "Invalid command usage."),
10      UNRECOGNIZED_ARGUMENTS(6, "Could not recognize some arguments."),
11      PROCESS_LOGIC_FAILED(7, "Exception raised in process logic."),
12      CONFIG_PARSING_FAILED(12, "Could not parse configuration values."),
13      COMMAND_LINE_CONFIG_READING_FAILED(13, "Could not read configuration from command line arguments."),
14      PRINT_VERSION_FAILED(14, "Could not print version."),
15      PRINT_CONFIG_SCHEMA_FAILED(15, "Could not print config schema."),
16      CONFIG_FILE_READING_FAILED(16, "Failed to load config file");
17  
18      private int code;
19      private String desc;
20  
21      GapaCommandLineExceptionType(int code, String desc) {
22          this.code = code;
23          this.desc = desc;
24      }
25  
26      @Override
27      public int getCode() {
28          return code;
29      }
30  
31      @Override
32      public String getDesc() {
33          return desc;
34      }
35  }
