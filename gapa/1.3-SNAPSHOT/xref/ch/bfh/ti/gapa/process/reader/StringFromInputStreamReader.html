3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.io.InputStreamReader;
6   import java.io.Reader;
7   import java.nio.charset.Charset;
8   
9   public class StringFromInputStreamReader {
10      public static String readStringFromInputStream(InputStream is, Charset charset, int buffSize) throws IOException {
11          char[] buf = new char[buffSize];
12          Reader r = new InputStreamReader(is, charset);
13          StringBuilder s = new StringBuilder();
14          int n;
15          while((n = r.read(buf)) >= 0)
16              s.append(buf, 0, n);
17          is.close();
18          return s.toString();
19      }
20  }
