3   import ch.bfh.ti.gapa.cli.config.reading.model.RawInput;
4   import org.json.JSONObject;
5   
6   /**
7    * Reads the data from a json object into
8    * an {@link RawInput} instance.
9    */
10  public class JsonReaderImpl implements JsonReader {
11  
12      /**
13       * Creates an instance of {@link RawInput} and copies data from a {@link JSONObject} to this instance.
14       * The websocketUri is read as string.
15       * The filters are optional and are read as {@link org.json.JSONArray}.
16       */
17      public void read(RawInput rawInput, JSONObject jsonObject) {
18          rawInput.setWebsocketUri(jsonObject.optString(ConfigJsonField.websocketUri.name()));
19          rawInput.setFilters(jsonObject.optJSONArray(ConfigJsonField.filters.name()));
20      }
21  }
