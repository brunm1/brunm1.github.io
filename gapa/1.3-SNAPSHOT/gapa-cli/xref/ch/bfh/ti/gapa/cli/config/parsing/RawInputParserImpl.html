3   import ch.bfh.ti.gapa.cli.config.reading.model.RawInput;
4   import ch.bfh.ti.gapa.cli.config.reading.model.RawInputField;
5   import ch.bfh.ti.gapa.process.interfaces.Input;
6   
7   import java.net.URI;
8   import java.net.URISyntaxException;
9   
10  /**
11   * Parses data in {@link RawInput} and writes the result into the given
12   * {@link Input} instance.
13   */
14  public class RawInputParserImpl implements RawInputParser {
15      private static URI convertStringToUri(String s) throws URISyntaxException {
16          return new URI(s);
17      }
18  
19      /**
20       * If a field in {@link RawInput} is not null, it is parsed and
21       * saved in {@link Input}. Existing values in {@link Input} are
22       * overwritten.
23       * @param rawInput Unparsed data collected from different sources
24       * @param input Target {@link Input} instance that will hold the parsed data
25       * @throws RawInputParseException when a {@link ch.bfh.ti.gapa.cli.config.reading.model.RawInputField} cannot be parsed
26       */
27      @Override
28      public void parse(RawInput rawInput, Input input) throws RawInputParseException {
29          if(rawInput.getWebsocketUri() != null) {
30              try {
31                  URI websocketUri = convertStringToUri(rawInput.getWebsocketUri());
32                  input.setWebsocketUri(websocketUri);
33              } catch (URISyntaxException e) {
34                  throw new RawInputParseException(RawInputField.websocketUri, e);
35              }
36          }
37  
38          if(rawInput.getFilters() != null) {
39              //was already validated, no further conversions needed
40              //filters are kept dynamic for maximum flexibility
41              //more validation on filters is done in the process layer
42              input.setFilters(rawInput.getFilters());
43          }
44      }
45  }
